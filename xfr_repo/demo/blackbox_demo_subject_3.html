Run Evaluation¶

import warnings
warnings.filterwarnings('ignore')
import numpy as np
np.random.seed(0)
import xfr.models.blackbox as bb

strise = bb.STRise(probe=probe,
                   refs=[ref],
                   gallery=[gallery],
                   black_box='resnetv6_pytorch')
strise.evaluate()
